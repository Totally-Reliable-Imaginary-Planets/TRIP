1//! # Planet module
2//! This module provides common definitions for planets and their associated types
3//! that need be used by a group to construct its own planet.
4//! The [Planet] struct is the **main component**: an instance of it represents the
5//! actual planet and contains all the logic and state (see [PlanetState]) needed to work as one, in fact
6//! this is what the orchestrator will interact with.
7//!
8//! You can instantiate a new planet by calling the [Planet::new] constructor method and passing
9//! valid construction parameters to it (look into its documentation to learn more).
10//!
11//! One of the construction parameters is a group-defined struct that implements the [PlanetAI] trait,
12//! which defines several methods for handling messages coming from the orchestrator and the explorers. This is
13//! the core of each group's planet implementation, as it defines the planet *behaviour*, that is
14//! how a planet "reacts" to the possible events or requests.
15//!
16//! ## Examples
17//! Intended usage (for planet definition, by groups):
18//!
19//! ```
20//! use std::sync::mpsc;
21//! use common_game::components::planet::{Planet, PlanetAI, PlanetState, PlanetType};
22//! use common_game::components::resource::{Combinator, Generator};
23//! use common_game::components::rocket::Rocket;
24//! use common_game::protocols::messages;
25//!
26//! // Group-defined AI struct
27//! struct AI { /* your AI state here */ };
28//!
29//! impl PlanetAI for AI {
30//!     fn handle_orchestrator_msg(
31//!         &mut self,
32//!         state: &mut PlanetState,
33//!         generator: &Generator,
34//!         combinator: &Combinator,
35//!         msg: messages::OrchestratorToPlanet
36//!     ) -> Option<messages::PlanetToOrchestrator> {
37//!         // your handler code here...
38//!         None
39//!     }
40//!
41//!     fn handle_explorer_msg(
42//!         &mut self,
43//!         state: &mut PlanetState,
44//!         generator: &Generator,
45//!         combinator: &Combinator,
46//!         msg: messages::ExplorerToPlanet
47//!     ) -> Option<messages::PlanetToExplorer> {
48//!         // your handler code here...
49//!         None
50//!     }
51//!
52//!     fn handle_asteroid(
53//!         &mut self,
54//!         state: &mut PlanetState,
55//!         generator: &Generator,
56//!         combinator: &Combinator,
57//!     ) -> Option<Rocket> {
58//!         // your handler code here...
59//!         None
60//!     }
61//!
62//!     fn start(&mut self, state: &PlanetState) { /* startup code */ }
63//!     fn stop(&mut self, state: &PlanetState) { /* stop code */ }
64//! }
65//!
66//! // This is the group's "export" function. It will be called by
67//! // the orchestrator to spawn your planet.
68//! pub fn create_planet(
69//!     rx_orchestrator: mpsc::Receiver<messages::OrchestratorToPlanet>,
70//!     tx_orchestrator: mpsc::Sender<messages::PlanetToOrchestrator>,
71//!     rx_explorer: mpsc::Receiver<messages::ExplorerToPlanet>,
72//!     tx_explorer: mpsc::Sender<messages::PlanetToExplorer>
73//! ) -> Planet {
74//!     let id = 1;
75//!     let ai = AI {};
76//!     let gen_rules = vec![/* your recipes */];
77//!     let comb_rules = vec![/* your recipes */];
78//!
79//!     // Construct the planet and return it
80//!     Planet::new(
81//!         id,
82//!         PlanetType::A,
83//!         Box::new(ai),
84//!         gen_rules,
85//!         comb_rules,
86//!         (rx_orchestrator, tx_orchestrator),
87//!         (rx_explorer, tx_explorer)
88//!     ).unwrap() // Don't call .unwrap()! You should do error checking instead.
89//! }
90//! ```

102/// The trait that defines the behaviour of a planet.
103///
104/// Structs implementing this trait are intended to be passed to the
105/// [Planet] constructor, so that the handlers can be invoked by the planet
106/// internal logic when certain messages are received on any of the planet channels.
107///
108/// The handlers can alter the planet state by accessing the
109/// `state` parameter, which is passed to the methods as a mutable borrow.
110/// A response can be sent by returning an optional message of the correct type,
111/// that will be forwarded to the associated channel passed on planet construction.
112pub trait PlanetAI: Send {
113    /// Handler for messages received by the orchestrator (receiving
114    /// end of the [OrchestratorToPlanet] channel).
115    /// The following messages will **not** invoke this handler:
116    /// - [OrchestratorToPlanet::StartPlanetAI] (see [PlanetAI::start])
117    /// - [OrchestratorToPlanet::StopPlanetAI] (see [PlanetAI::stop])
118    /// - [OrchestratorToPlanet::Asteroid] (see [PlanetAI::handle_asteroid])
119    ///
120    /// Check [PlanetAI] docs for general meaning of the parameters and return type.
121    fn handle_orchestrator_msg(
122        &mut self,
123        state: &mut PlanetState,
124        generator: &Generator,
125        combinator: &Combinator,
126        msg: OrchestratorToPlanet,
127    ) -> Option<PlanetToOrchestrator>;
128
129    /// Handler for **all** messages received by an explorer (receiving
130    /// end of the [ExplorerToPlanet] channel).
131    ///
132    /// Check [PlanetAI] docs for general meaning of the parameters and return type.
133    fn handle_explorer_msg(
134        &mut self,
135        state: &mut PlanetState,
136        generator: &Generator,
137        combinator: &Combinator,
138        msg: ExplorerToPlanet,
139    ) -> Option<PlanetToExplorer>;
140
141    /// This handler will be invoked when a [OrchestratorToPlanet::Asteroid]
142    /// message is received. It's important to handle *Asteroid* messages
143    /// correctly, as this will the determine the planet survival.
144    ///
145    /// # Returns
146    /// In order to survive, an owned [Rocket] **must** be returned from this method;
147    /// if `None` is returned instead, the planet will (or *should*) be **destroyed** by the orchestrator
148    fn handle_asteroid(
149        &mut self,
150        state: &mut PlanetState,
151        generator: &Generator,
152        combinator: &Combinator,
153    ) -> Option<Rocket>;
154
155    /// This method will be invoked when a [OrchestratorToPlanet::StartPlanetAI]
156    /// is received, but **only if** the planet is currently in a *stopped* state.
157    ///
158    /// Start messages received when planet is already running are **ignored**.
159    fn start(&mut self, state: &PlanetState);
160
161    /// This method will be invoked when a [OrchestratorToPlanet::StopPlanetAI]
162    /// is received, but **only if** the planet is currently in a *running* state.
163    ///
164    /// Stop messages received when planet is already stopped are **ignored**.
165    fn stop(&mut self, state: &PlanetState);
166}

168// Defines the planet rules constraints
169struct PlanetConstraints {
170    n_energy_cells: usize,
171    unbounded_gen_rules: bool,
172    can_have_rocket: bool,
173    n_comb_rules: usize,
174}

176/// Planet types definitions, intended to be passed
177/// to the planet constructor. Identifies the planet rules constraints,
178/// with each type having its own.
179#[derive(Debug, Clone, Copy)]
180pub enum PlanetType {
181    A,
182    B,
183    C,
184    D,
185}
186
187impl PlanetType {
188    const N_ENERGY_CELLS: usize = 5;
189    const N_RESOURCE_COMB_RULES: usize = 6;
190
191    // Returns a tuple with the constraints associated with the planet type,
192    // as described in the project specifications.
193    fn constraints(&self) -> PlanetConstraints {
194        match self {
195            PlanetType::A => PlanetConstraints {
196                n_energy_cells: Self::N_ENERGY_CELLS,
197                unbounded_gen_rules: false,
198                can_have_rocket: true,
199                n_comb_rules: 0,
200            },
201            PlanetType::B => PlanetConstraints {
202                n_energy_cells: 1,
203                unbounded_gen_rules: true,
204                can_have_rocket: false,
205                n_comb_rules: 1,
206            },
207            PlanetType::C => PlanetConstraints {
208                n_energy_cells: 1,
209                unbounded_gen_rules: false,
210                can_have_rocket: true,
211                n_comb_rules: Self::N_RESOURCE_COMB_RULES,
212            },
213            PlanetType::D => PlanetConstraints {
214                n_energy_cells: Self::N_ENERGY_CELLS,
215                unbounded_gen_rules: true,
216                can_have_rocket: false,
217                n_comb_rules: 0,
218            },
219        }
220    }
221}

223/// This struct is a representation of the internal state
224/// of the planet. Through its public methods, it gives access to the all resources
225/// of the planet:
226/// - A vec of [EnergyCell].
227/// - An optional [Rocket], that can be built accordingly to the planet type.
228/// - [Generator] for generating basic resources.
229/// - [Combinator] for combining basic resources into complex ones.
230pub struct PlanetState {
231    id: u32,
232    energy_cells: Vec<EnergyCell>,
233    rocket: Option<Rocket>,
234    can_have_rocket: bool,
235}
236
237impl PlanetState {
238    /// Returns the planet id.
239    pub fn id(&self) -> u32 {
240        self.id
241    }
242
243    /// Indexed getter accessor for the [EnergyCell] vec.
244    ///
245    /// # Returns
246    /// An immutable borrow of the *i-th* energy cell.
247    ///
248    /// # Panics
249    /// This method will panic if the index `i` is out of bounds.
250    /// Always check the number of energy cells available with [PlanetState::cells_count].
251    pub fn cell(&self, i: usize) -> &EnergyCell {
252        &self.energy_cells[i]
253    }
254
255    /// Indexed *mutable* getter accessor for the [EnergyCell] vec.
256    ///
257    /// # Returns
258    /// An mutable borrow of the *i-th* energy cell.
259    ///
260    /// # Panics
261    /// This method will panic if the index `i` is out of bounds.
262    /// Always check the number of energy cells available with [PlanetState::cells_count].
263    pub fn cell_mut(&mut self, i: usize) -> &mut EnergyCell {
264        &mut self.energy_cells[i]
265    }
266
267    /// Returns the number of energy cells owned by
268    /// the planet. This is the actual size of the internal
269    /// vec containing the cells.
270    pub fn cells_count(&self) -> usize {
271        self.energy_cells.len()
272    }
273
274    /// Returns an *immutable* iterator over the energy cells owned by the planet.
275    pub fn cells_iter(&self) -> Iter<'_, EnergyCell> {
276        self.energy_cells.iter()
277    }
278
279    /// Returns a *mutable* iterator over the energy cells owned by the planet.
280    pub fn cells_iter_mut(&mut self) -> IterMut<'_, EnergyCell> {
281        self.energy_cells.iter_mut()
282    }
283
284    /// Charges the first empty (discharged) cell.
285    /// Returns an optional [Sunray] if there's no cell to charge.
286    pub fn charge_cell(&mut self, sunray: Sunray) -> Option<Sunray> {
287        match self.empty_cell() {
288            None => Some(sunray),
289            Some((cell, _)) => {
290                cell.charge(sunray);
291                None
292            }
293        }
294    }
295
296    /// Returns a tuple containing a *mutable* borrow of the first empty (discharged) cell
297    /// and its index, or `None` if there isn't any.
298    pub fn empty_cell(&mut self) -> Option<(&mut EnergyCell, usize)> {
299        let idx = self.energy_cells.iter().position(|cell| !cell.is_charged());
300        idx.map(|i| (&mut self.energy_cells[i], i))
301    }
302
303    /// Returns a tuple containing a *mutable* borrow of the first full (charged) cell
304    /// and its index, or `None` if there isn't any.
305    pub fn full_cell(&mut self) -> Option<(&mut EnergyCell, usize)> {
306        let idx = self.energy_cells.iter().position(|cell| cell.is_charged());
307        idx.map(|i| (&mut self.energy_cells[i], i))
308    }
309
310    /// Returns `true` if the planet can have a rocket.
311    pub fn can_have_rocket(&self) -> bool {
312        self.can_have_rocket
313    }
314
315    /// Returns `true` if the planet has a rocket built and ready to launch.
316    pub fn has_rocket(&self) -> bool {
317        self.rocket.is_some()
318    }
319
320    /// Takes the rocket out of the planet state (if there is one), leaving
321    /// `None` in its place.
322    pub fn take_rocket(&mut self) -> Option<Rocket> {
323        self.rocket.take()
324    }
325
326    /// Constructs a rocket using the *i-th* [EnergyCell] of the planet and stores it
327    /// inside the planet, taking ownership of it.
328    ///
329    /// # Panics
330    /// This method will panic if the index `i` is out of bounds.
331    /// Always check the number of energy cells available with [PlanetState::cells_count].
332    ///
333    /// # Errors
334    /// Returns an error if:
335    /// - The planet type prohibits the storing of rockets.
336    /// - The planet already has a rocket built.
337    /// - The energy cell is not charged
338    pub fn build_rocket(&mut self, i: usize) -> Result<(), String> {
339        if !self.can_have_rocket {
340            Err("This planet type can't have rockets.".to_string())
341        } else if self.has_rocket() {
342            Err("This planet already has a rocket.".to_string())
343        } else {
344            let energy_cell = self.cell_mut(i);
345            Rocket::new(energy_cell).map(|rocket| {
346                self.rocket = Some(rocket);
347            })
348        }
349    }
350}

352/// Main, top-level planet definition. This type is built on top of
353/// [PlanetState], [PlanetType] and [PlanetAI], through composition.
354///
355/// It needs to be constructed by each group as it represents the actual planet
356/// and contains the base logic that runs the AI. Also, this is what should be
357/// returned to the orchestrator.
358///
359/// See module-level docs for more general info.
360pub struct Planet {
361    state: PlanetState,
362    planet_type: PlanetType,
363    pub ai: Box<dyn PlanetAI>,
364    generator: Generator,
365    combinator: Combinator,
366
367    from_orchestrator: mpsc::Receiver<OrchestratorToPlanet>,
368    to_orchestrator: mpsc::Sender<PlanetToOrchestrator>,
369    from_explorer: mpsc::Receiver<ExplorerToPlanet>,
370    to_explorer: mpsc::Sender<PlanetToExplorer>,
371}
372
373impl Planet {
374    /// Constructor for the [Planet] type.
375    ///
376    /// # Errors
377    /// Returns an error if the construction parameters are *invalid* (they violate the `planet_type` constraints).
378    ///
379    /// # Arguments
380    /// - `id` - The identifier to assign to the planet.
381    /// - `planet_type` - Type of the planet. Constraints the rules of the planet.
382    /// - `ai` - A group-defined struct implementing the [PlanetAI] trait.
383    /// - `gen_rules` - A vec of [BasicResourceType] containing the basic resources the planet will be able to generate.
384    /// - `comb_rules` - A vec of [ComplexResourceType] containing the complex resources the planet will be able to make.
385    /// - `orchestrator_channels` - A pair containing the receiver and sender half
386    ///   of the channels [OrchestratorToPlanet] and [PlanetToOrchestrator].
387    /// - `explorer_channels` - A pair containing the [mpsc::Receiver] and [mpsc::Sender] half
388    ///   of the channels [ExplorerToPlanet] and [PlanetToExplorer].
389    pub fn new(
390        id: u32,
391        planet_type: PlanetType,
392        ai: Box<dyn PlanetAI>,
393        gen_rules: Vec<BasicResourceType>,
394        comb_rules: Vec<ComplexResourceType>,
395        orchestrator_channels: (
396            mpsc::Receiver<OrchestratorToPlanet>,
397            mpsc::Sender<PlanetToOrchestrator>,
398        ),
399        explorer_channels: (
400            mpsc::Receiver<ExplorerToPlanet>,
401            mpsc::Sender<PlanetToExplorer>,
402        ),
403    ) -> Result<Planet, String> {
404        let PlanetConstraints {
405            n_energy_cells,
406            unbounded_gen_rules,
407            can_have_rocket,
408            n_comb_rules,
409        } = planet_type.constraints();
410        let (from_orchestrator, to_orchestrator) = orchestrator_channels;
411        let (from_explorer, to_explorer) = explorer_channels;
412
413        if gen_rules.is_empty() {
414            Err("gen_rules is empty".to_string())
415        } else if !unbounded_gen_rules && gen_rules.len() > 1 {
416            Err(format!(
417                "Too many generation rules (Planet type {:?} is limited to 1)",
418                planet_type
419            ))
420        } else if comb_rules.len() > n_comb_rules {
421            Err(format!(
422                "Too many combination rules (Planet type {:?} is limited to {})",
423                planet_type, n_comb_rules
424            ))
425        } else {
426            let mut generator = Generator::new();
427            let mut combinator = Combinator::new();
428
429            // add gen and comb rules to the planet generator and combinator
430            for r in gen_rules {
431                let _ = generator.add(r);
432            }
433            for r in comb_rules {
434                let _ = combinator.add(r);
435            }
436
437            Ok(Planet {
438                state: PlanetState {
439                    id,
440                    energy_cells: (0..n_energy_cells).map(|_| EnergyCell::new()).collect(),
441                    can_have_rocket,
442                    rocket: None,
443                },
444                planet_type,
445                ai,
446                generator,
447                combinator,
448                from_orchestrator,
449                to_orchestrator,
450                from_explorer,
451                to_explorer,
452            })
453        }
454    }
455
456    /// Starts the planet in a *stopped* state, waiting for a [OrchestratorToPlanet::StartPlanetAI],
457    /// then invokes [PlanetAI::start] and runs the main message polling loop.
458    /// See [PlanetAI] docs to know more about when handlers are invoked and how the planet reacts
459    /// to the different messages.
460    ///
461    /// This method is *blocking* and should be called by the orchestrator in a separate thread.
462    ///
463    /// # Errors
464    /// This method should not return, but if it does, it returns an error message
465    /// describing the cause. For now, the only possible cause of error is orchestrator disconnection
466    /// from one of the channels.
467    pub fn run(&mut self) -> Result<(), String> {
468        // run the planet stopped by default
469        // and wait for a StartPlanetAI message
470        self.wait_for_start()?;
471
472        self.ai.start(&self.state);
473
474        // maybe spawn a thread for async event handling ?
475        loop {
476            // TODO: disconnection error handling
477
478            // orchestrator incoming message polling
479            match self.from_orchestrator.try_recv() {
480                // TODO: do something with the StartPlanetAI message content
481                Ok(OrchestratorToPlanet::StartPlanetAI) => {}
482                // TODO: do something with the StopPlanetAI message content
483                Ok(OrchestratorToPlanet::StopPlanetAI) => {
484                    self.ai.stop(&self.state);
485                    self.wait_for_start()?; // blocking wait
486
487                    // restart AI
488                    self.ai.start(&self.state)
489                }
490                Ok(OrchestratorToPlanet::Asteroid(_)) => {
491                    let rocket =
492                        self.ai
493                            .handle_asteroid(&mut self.state, &self.generator, &self.combinator);
494                    self.to_orchestrator
495                        .send(PlanetToOrchestrator::AsteroidAck {
496                            planet_id: self.id(),
497                            rocket,
498                        })
499                        .map_err(|_| "Orchestrator disconnected".to_string())?;
500                }
501                Ok(msg) => {
502                    self.ai
503                        .handle_orchestrator_msg(
504                            &mut self.state,
505                            &self.generator,
506                            &self.combinator,
507                            msg,
508                        )
509                        .map(|response| self.to_orchestrator.send(response))
510                        .transpose()
511                        .map_err(|_| "Orchestrator disconnected".to_string())?;
512                }
513
514                Err(mpsc::TryRecvError::Disconnected) => {
515                    return Err("Orchestrator disconnected".to_string());
516                }
517                Err(mpsc::TryRecvError::Empty) => {}
518            }
519
520            // explorer incoming message polling
521            match self.from_explorer.try_recv() {
522                Ok(msg) => {
523                    if let Some(response) = self.ai.handle_explorer_msg(
524                        &mut self.state,
525                        &self.generator,
526                        &self.combinator,
527                        msg,
528                    ) {
529                        self.to_explorer
530                            .send(response)
531                            .unwrap_or_else(|_| println!("No explorer connected!"))
532                    }
533                }
534
535                Err(mpsc::TryRecvError::Disconnected) => {}
536                Err(mpsc::TryRecvError::Empty) => {}
537            }
538        }
539    }

541    // private helper function that blocks until
542    // a StartPlanetAI message is received
543    fn wait_for_start(&self) -> Result<(), String> {
544        loop {
545            // TODO: error handling
546            let recv_re = self.from_orchestrator.recv();
547            match recv_re {
548                // TODO: do something with the StartPlanetAI message content
549                Ok(OrchestratorToPlanet::StartPlanetAI) => return Ok(()),
550                Err(_) => return Err("Orchestrator disconnected!".to_string()),
551                _ => {}
552            }
553        }
554    }

790    #[test]
791    fn test_planet_run_loop_survival() {
792        // 1. Orch -> Planet
793        let (tx_to_planet_orch, rx_from_orch) = mpsc::channel::<OrchestratorToPlanet>();
794        // 2. Planet -> Orch
795        let (tx_from_planet_orch, rx_to_orch) = mpsc::channel::<PlanetToOrchestrator>();
796
797        // 3. Expl -> Planet
798        let (_tx_to_planet_expl, rx_from_expl) = mpsc::channel::<ExplorerToPlanet>();
799        // 4. Planet -> Expl
800        let (tx_from_planet_expl, _rx_from_planet) = mpsc::channel::<PlanetToExplorer>();
801
802        // Build Planet
803        let mut planet = Planet::new(
804            100,
805            PlanetType::A,
806            Box::new(MockAI::new()),
807            vec![BasicResourceType::Oxygen],
808            vec![],
809            (rx_from_orch, tx_from_planet_orch),
810            (rx_from_expl, tx_from_planet_expl),
811        )
812        .expect("Failed to create planet");
813
814        // Spawn thread
815        let handle = thread::spawn(move || {
816            let _ = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
817                let _ = planet.run();
818            }));
819        });
820
821        // 1. Start AI
822        tx_to_planet_orch
823            .send(OrchestratorToPlanet::StartPlanetAI)
824            .unwrap();
825        thread::sleep(Duration::from_millis(50));
826
827        // 2. Send Sunray
828        tx_to_planet_orch
829            .send(OrchestratorToPlanet::Sunray(Sunray::new()))
830            .unwrap();
831
832        // Expect Ack
833        if let Ok(PlanetToOrchestrator::SunrayAck { planet_id, .. }) =
834            rx_to_orch.recv_timeout(Duration::from_millis(200))
835        {
836            assert_eq!(planet_id, 100);
837        } else {
838            panic!("Did not receive SunrayAck");
839        }
840
841        // 3. Send Asteroid (AI should build rocket using the charged cell)
842        tx_to_planet_orch
843            .send(OrchestratorToPlanet::Asteroid(Asteroid::new()))
844            .unwrap();
845
846        // 4. Expect Survival (Ack with Some(Rocket))
847        match rx_to_orch.recv_timeout(Duration::from_millis(200)) {
848            Ok(PlanetToOrchestrator::AsteroidAck {
849                planet_id, rocket, ..
850            }) => {
851                assert_eq!(planet_id, 100);
852                assert!(rocket.is_some(), "Planet failed to build rocket!");
853            }
854            Ok(_) => panic!("Wrong message type"),
855            Err(_) => panic!("Timeout waiting for AsteroidAck"),
856        }
857
858        // 5. Stop
859        tx_to_planet_orch
860            .send(OrchestratorToPlanet::StopPlanetAI)
861            .unwrap();
862
863        drop(tx_to_planet_orch);
864        let _ = handle.join();
865    }

868    fn test_resource_creation() {
869        let (orch_ch, expl_ch) = get_test_channels();
870        let gen_rules = vec![BasicResourceType::Oxygen, BasicResourceType::Hydrogen];
871        let comb_rules = vec![ComplexResourceType::Water];
872        let mut planet = Planet::new(
873            0,
874            PlanetType::B,
875            Box::new(MockAI::new()),
876            gen_rules,
877            comb_rules,
878            orch_ch,
879            expl_ch,
880        )
881        .unwrap();
882
883        // aliases for planet internals
884        let state = &mut planet.state;
885        let generator = &planet.generator;
886        let combinator = &planet.combinator;
887
888        // gen oxygen
889        let cell = state.cell_mut(0);
890        cell.charge(Sunray::new());
891
892        let oxygen = generator.make_oxygen(cell);
893        assert!(oxygen.is_ok());
894        let oxygen = oxygen.unwrap();
895
896        // gen hydrogen
897        let cell = state.cell_mut(0);
898        cell.charge(Sunray::new());
899
900        let hydrogen = generator.make_hydrogen(cell);
901        assert!(hydrogen.is_ok());
902        let hydrogen = hydrogen.unwrap();
903
904        // combine the two elements into water
905        let cell = state.cell_mut(0);
906        cell.charge(Sunray::new());
907
908        let diamond = combinator.make_water(hydrogen, oxygen, cell);
909        assert!(diamond.is_ok());
