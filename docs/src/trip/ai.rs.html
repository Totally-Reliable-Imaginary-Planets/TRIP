1//! AI logic for our Planet implementation.
2//!
3//! This module defines the [`AI`] struct, which implements the [`PlanetAI`]
4//! trait from `common_game`. The AI controls how our planet reacts to
5//! orchestrator messages, explorer requests, and asteroid events.
6//!
7//! # Overview
8//!
9//! The AI manages three major responsibilities:
10//!
11//! 1. **Lifecycle control** via `start()` and `stop()`.
12//!    - When stopped, the AI rejects all messages and produces no output.
13//! 2. **Message handling**
14//!    - [`handle_orchestrator_msg`] processes messages from the orchestrator,
15//!      including sunrays, internal state requests, and others.
16//!    - [`handle_explorer_msg`] processes queries and requests from explorers
17//!      related to energy, basic resources, supported combinations, and complex
18//!      combinations.
19//! 3. **Asteroid response logic**
20//!    - [`handle_asteroid`] launches an existing rocket or attempts to build
21//!      and launch a new one.
22//!
23//! # AI Runtime Model
24//!
25//! The AI maintains an internal `running: bool` flag.
26//! - When `running == false`, the planet is considered inactive and **all
27//!   incoming messages are ignored**.
28//! - The orchestrator controls this state via `StartPlanetAI` and
29//!   `StopPlanetAI` messages.
30//!
31//! The planet never blocks inside the AI; blocking occurs only in the
32//! outer planet loop that receives messages from channels.
33//!
34//! # Supported Features
35//!
36//! The AI supports:
37//! - **Sunray absorption and energy cell charging**
38//! - **Rocket construction via charged cells**
39//! - **Internal state reporting**
40//! - **Basic resource handling for Oxygen**
41//! - **Fallback error reporting for unsupported combinations**
42//! - **Asteroid-triggered rocket launching**
43//!
44//! # Unsupported Features (as of current version)
45//!
46//! The following message types are acknowledged but **not implemented** and
47//! return `None` (or panic if explicitly marked with `todo!()` in the code):
48//!
49//! - Incoming and outgoing explorer routing requests
50//! - Complex resource generation beyond the Oxygen path
51//! - Planet kill event (currently ignored; real implementation should finalize
52//!   the planet's lifecycle)
53//!
54//! # Thread Safety and Side Effects
55//!
56//! - The AI mutates [`PlanetState`] extensively (charging cells, building and
57//!   launching rockets, creating resources).
58//! - Logging is performed using the `log` crate.
59//! - No global state is modified, and the struct is `Send` + `Sync` via its
60//!   field structure.
61//!
62//! # Protocol Guarantees
63//!
64//! This implementation respects the project protocol by:
65//! - Never reading from channels directly.
66//! - Producing a response only when required.
67//! - Logging all relevant state transitions.
68//! - Maintaining deterministic behavior (no randomness here).
69//!
70//! # See Also
71//!
72//! - [`PlanetState`]
73//! - [`Generator`]
74//! - [`Combinator`]
75//! - [`PlanetAI` trait](common_game::components::planet::PlanetAI)
76

89/// AI implementation for our planet.
90///
91/// This AI governs message handling, lifecycle control, energy management,
92/// rocket building, resource generation, and asteroid defense.
93///
94/// See the module-level documentation for full details.
95pub(crate) struct AI {
96    running: bool,
97}
98
99impl AI {
100    /// Creates a new, inactive [`AI`] instance.
101    ///
102    /// The AI begins in the `running = false` state, meaning no incoming
103    /// messages will be processed until [`start`](PlanetAI::start) is called.
104    pub(crate) fn new() -> Self {
105        Self { running: false }
106    }
107
108    /// Returns `true` if the AI is currently active, otherwise logs that the
109    /// AI ignored a message due to being stopped and returns `false`.
110    ///
111    /// # Parameters
112    /// - `planet_id`: The ID of the planet for contextual logging.
113    ///
114    /// # Returns
115    /// `true` if `running == true`, `false` otherwise.
116    ///
117    /// # Side Effects
118    /// - Writes a debug log message when inactive.
119    fn is_running(&self, planet_id: u32) -> bool {
120        if !self.running {
121            debug!("planet_id={planet_id} msg_ignored: ai_stopped");
122            return false;
123        }
124        true
125    }
126
127    /// Transforms a [`ComplexResourceRequest`] into a pair of [`GenericResource`]
128    /// values suitable for error reporting or unsupported-combination responses.
129    ///
130    /// This function does not validate whether the combination is allowed on
131    /// this planet; it only decomposes the request into its constituent parts.
132    ///
133    /// # Returns
134    /// A pair `(left, right)` representing the two logical inputs to a
135    /// combination request.
136    fn get_generic_resources(msg: ComplexResourceRequest) -> (GenericResource, GenericResource) {
137        match msg {
138            ComplexResourceRequest::Water(h, o) => (
139                GenericResource::BasicResources(BasicResource::Hydrogen(h)),
140                GenericResource::BasicResources(BasicResource::Oxygen(o)),
141            ),
142            ComplexResourceRequest::Diamond(c1, c2) => (
143                GenericResource::BasicResources(BasicResource::Carbon(c1)),
144                GenericResource::BasicResources(BasicResource::Carbon(c2)),
145            ),
146            ComplexResourceRequest::Life(w, c) => (
147                GenericResource::ComplexResources(ComplexResource::Water(w)),
148                GenericResource::BasicResources(BasicResource::Carbon(c)),
149            ),
150            ComplexResourceRequest::Robot(s, l) => (
151                GenericResource::BasicResources(BasicResource::Silicon(s)),
152                GenericResource::ComplexResources(ComplexResource::Life(l)),
153            ),
154            ComplexResourceRequest::Dolphin(w, l) => (
155                GenericResource::ComplexResources(ComplexResource::Water(w)),
156                GenericResource::ComplexResources(ComplexResource::Life(l)),
157            ),
158            ComplexResourceRequest::AIPartner(r, d) => (
159                GenericResource::ComplexResources(ComplexResource::Robot(r)),
160                GenericResource::ComplexResources(ComplexResource::Diamond(d)),
161            ),
162        }
163    }
164
165    /// Handles a [`Sunray`] by charging the first uncharged energy cell and
166    /// attempting to build a rocket on that cell.
167    ///
168    /// This method encapsulates the sunray-handling logic used by
169    /// [`handle_orchestrator_msg`](PlanetAI::handle_orchestrator_msg).
170    ///
171    /// # Behavior
172    /// - Charges the first available uncharged cell.
173    /// - Attempts to build a rocket on that cell; logs success or failure.
174    /// - Logs relevant diagnostic information.
175    ///
176    /// # Side Effects
177    /// - Mutates the [`PlanetState`] (cell charge, rocket construction).
178    /// - Emits debug, info, or error logs.
179    fn handle_sunray(state: &mut PlanetState, s: Sunray) {
180        debug!("planet_id={} incoming_sunray", state.id());
181        if let Some(index) = state.cells_iter().position(|cell| !cell.is_charged()) {
182            let cell = state.cell_mut(index);
183            cell.charge(s);
184            debug!("planet_id={} sunray: charging cell", state.id());
185            match state.build_rocket(index) {
186                Ok(()) => info!("planet_id={} rocket_built", state.id()),
187                Err(e) => warn!("planet_id={} rocket_build_failed: {}", state.id(), e),
188            }
189        } else {
190            warn!("planet_id={} sunray: no_uncharged_cells", state.id());
191        }
192        debug!("planet_id={} outgoing_sunray_ack", state.id());
193    }
194}

196impl PlanetAI for AI {
197    /// Activates the AI and enables message processing.
198    ///
199    /// Called by the planet runtime when initialization completes.
200    /// After this call, incoming messages to the AI will be processed normally.
201    ///
202    /// # Side Effects
203    /// - Sets `running = true`
204    /// - Logs an informational `ai_started` message
205    fn on_start(&mut self, state: &PlanetState, _: &Generator, _: &Combinator) {
206        self.running = true;
207        info!("planet_id={} ai_started", state.id());
208    }
209
210    /// Deactivates the AI and stops all message processing.
211    ///
212    /// All message handlers will return `None` until the AI is restarted.
213    ///
214    /// # Side Effects
215    /// - Sets `running = false`
216    /// - Logs an informational `ai_stopped` message
217    fn on_stop(&mut self, state: &PlanetState, _: &Generator, _: &Combinator) {
218        self.running = false;
219        info!("planet_id={} ai_stopped", state.id());
220    }
221
222    /// Handles a sunray by delegating to the internal charging logic.
223    ///
224    /// # Behavior
225    /// - Consumes the incoming sunray to charge the first available energy cell.
226    /// - Attempts to build a rocket immediately after charging.
227    /// - This is a wrapper around the static [`AI::handle_sunray`] method.
228    fn handle_sunray(&mut self, state: &mut PlanetState, _: &Generator, _: &Combinator, s: Sunray) {
229        if self.is_running(state.id()) {
230            AI::handle_sunray(state, s);
231        }
232    }
233
234    /// Provides a `DummyPlanetState` object representing the current planet state.
235    ///
236    /// # Behavior
237    /// - Converts the current `PlanetState` into a `DummyPlanetState`.
238    ///
239    /// # Returns
240    /// A `DummyPlanetState` representing the current state of the planet.
241    fn handle_internal_state_req(
242        &mut self,
243        state: &mut PlanetState,
244        _: &Generator,
245        _: &Combinator,
246    ) -> DummyPlanetState {
247        state.to_dummy()
248    }
249
250    /// Handles messages from an explorer interacting with this planet.
251    ///
252    /// The AI responds to explorer queries about:
253    /// - Supported basic resources
254    /// - Supported combination rules
255    /// - Energy availability
256    /// - Requests to generate Oxygen
257    ///
258    /// Unsupported combinations or unsupported resource requests result in
259    /// `None` or an appropriate error response.
260    ///
261    /// # Behavior
262    ///
263    /// - If the AI is stopped, returns `None`.
264    /// - Basic resource generation is supported only for Oxygen.
265    /// - Combination attempts always produce an `Err` payload indicating
266    ///   unsupported functionality.
267    ///
268    /// # Returns
269    /// - `Some(response)` if a valid response exists.
270    /// - `None` if the AI is stopped or if the request cannot be fulfilled.    
271    fn handle_explorer_msg(
272        &mut self,
273        state: &mut PlanetState,
274        generator: &Generator,
275        comb: &Combinator,
276        msg: ExplorerToPlanet,
277    ) -> Option<PlanetToExplorer> {
278        if !self.is_running(state.id()) {
279            return None;
280        }
281        match msg {
282            ExplorerToPlanet::SupportedResourceRequest { explorer_id } => {
283                debug!(
284                    "planet_id={} explorer_id={} outgoing_supported_resource_response",
285                    state.id(),
286                    explorer_id
287                );
288                Some(PlanetToExplorer::SupportedResourceResponse {
289                    resource_list: generator.all_available_recipes(),
290                })
291            }
292            ExplorerToPlanet::GenerateResourceRequest {
293                explorer_id,
294                resource: BasicResourceType::Oxygen,
295            } => state
296                .cells_iter()
297                .position(EnergyCell::is_charged)
298                .and_then(|index| generator.make_oxygen(state.cell_mut(index)).ok())
299                .map(|r| {
300                    debug!(
301                        "planet_id={} explorer_id={} generate_oxygen: success",
302                        state.id(),
303                        explorer_id
304                    );
305                    PlanetToExplorer::GenerateResourceResponse {
306                        resource: Some(common_game::components::resource::BasicResource::Oxygen(r)),
307                    }
308                })
309                .or_else(|| {
310                    warn!(
311                        "planet_id={} explorer_id={} generate_oxygen: failed",
312                        state.id(),
313                        explorer_id
314                    );
315                    None
316                }),
317            ExplorerToPlanet::GenerateResourceRequest { explorer_id, .. } => {
318                debug!(
319                    "planet_id={} explorer_id={} generate_resource: unsupported",
320                    state.id(),
321                    explorer_id
322                );
323                None
324            }
325            ExplorerToPlanet::SupportedCombinationRequest { explorer_id, .. } => {
326                debug!(
327                    "planet_id={} explorer_id={} outgoing_supported_combination_response",
328                    state.id(),
329                    explorer_id
330                );
331                Some(PlanetToExplorer::SupportedCombinationResponse {
332                    combination_list: comb.all_available_recipes(),
333                })
334            }
335            ExplorerToPlanet::CombineResourceRequest { explorer_id, msg } => {
336                debug!(
337                    "planet_id={} explorer_id={} incoming_combine_request: {:?}",
338                    state.id(),
339                    explorer_id,
340                    msg
341                );
342                let (left, right) = AI::get_generic_resources(msg);
343                debug!(
344                    "planet_id={} explorer_id={} outgoing_combine_response=unsupported_combination",
345                    state.id(),
346                    explorer_id
347                );
348                Some(PlanetToExplorer::CombineResourceResponse {
349                    complex_response: Err(("unsupported_combination".to_string(), left, right)),
350                })
351            }
352            ExplorerToPlanet::AvailableEnergyCellRequest { explorer_id } => {
353                let tmp = state.cells_iter().filter(|&cell| cell.is_charged()).count();
354                let count = tmp.try_into().unwrap_or_default();
355                debug!(
356                    "planet_id={} explorer_id={} outgoing_energy_cell_count={}",
357                    state.id(),
358                    explorer_id,
359                    count
360                );
361                Some(PlanetToExplorer::AvailableEnergyCellResponse {
362                    available_cells: count,
363                })
364            }
365        }
366    }
367
368    /// Handles an asteroid impact event.
369    ///
370    /// # Behavior
371    ///
372    /// - If a rocket already exists in the state, it is launched immediately.
373    /// - Otherwise, the AI searches for the first charged energy cell and
374    ///   attempts to build a rocket on it.
375    /// - If rocket construction succeeds, the rocket is launched.
376    /// - If construction fails or no charged cell exists, `None` is returned.
377    ///
378    /// # Side Effects
379    /// - Mutates the planet state by consuming energy cells and creating rockets.
380    /// - Logs informational or warning messages depending on outcome.
381    ///
382    /// # Returns
383    /// `Some(Rocket)` if a rocket is launched, otherwise `None`.    
384    fn handle_asteroid(
385        &mut self,
386        state: &mut PlanetState,
387        _: &Generator,
388        _: &Combinator,
389    ) -> Option<Rocket> {
390        if !self.is_running(state.id()) {
391            return None;
392        }
393        if state.has_rocket() {
394            info!(
395                "planet_id={} asteroid_event: existing_rocket_launched",
396                state.id()
397            );
398            return state.take_rocket();
399        }
400        if let Some(index) = state.cells_iter().position(EnergyCell::is_charged) {
401            match state.build_rocket(index) {
402                Ok(()) => {
403                    info!(
404                        "planet_id={} asteroid_event: rocket_built_and_launched",
405                        state.id()
406                    );
407                    return state.take_rocket();
408                }
409                Err(e) => error!(
410                    "planet_id={} asteroid_event: rocket_build_failed {}",
411                    state.id(),
412                    e
413                ),
414            }
415        } else {
416            warn!(
417                "planet_id={} asteroid_event: no_charged_cells_available",
418                state.id()
419            );
420        }
421        None
422    }
423}
