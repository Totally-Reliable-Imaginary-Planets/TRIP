34    /// Handles a message from the orchestrator.
35    ///
36    /// This method processes incoming messages from the orchestrator when the planet is active.
37    /// If the planet is stopped (`self.is_stopped`), no messages are processed and `None` is returned immediately.
38    ///
39    /// # Behavior by Message Type
40    ///
41    /// - [`OrchestratorToPlanet::Sunray(s)`]:
42    ///   - Finds the first uncharged cell and charges it with the sunray data.
43    ///   - Attempts to build a rocket on that cell.
44    ///   - Always returns a [`SunrayAck`] containing the planet ID.
45    ///
46    /// - [`OrchestratorToPlanet::IncomingExplorerRequest`], [`OrchestratorToPlanet::OutgoingExplorerRequest`],
47    ///   [`OrchestratorToPlanet::InternalStateRequest`]:
48    ///   - Marked with `todo!()` â€” these will panic in release and should be implemented.
49    ///
50    /// - [`OrchestratorToPlanet::Asteroid`], [`OrchestratorToPlanet::StartPlanetAI`], [`OrchestratorToPlanet::StopPlanetAI`]:
51    ///   - Silently ignored (`None` returned).
52    ///
53    /// # Returns
54    ///
55    /// - `Some(PlanetToOrchestrator)`: A response is generated.
56    /// - `None`: No response is sent, either because the planet is stopped or the message is ignored.
57    ///
58    /// # Logging
59    ///
60    /// # Panics
61    ///
62    /// Panics if:
63    /// - An unimplemented message variant (`IncomingExplorerRequest`, etc.) is received.
64    ///
65    /// # See Also
66    ///
67    /// - [`PlanetState::build_rocket`]
68    /// - [`SunrayAck`]
69    fn handle_orchestrator_msg(
70        &mut self,
71        state: &mut PlanetState,
72        _: &Generator,
73        _: &Combinator,
74        msg: OrchestratorToPlanet,
75    ) -> Option<PlanetToOrchestrator> {
76        if self.is_stopped {
77            return None;
78        }
79        match msg {
80            OrchestratorToPlanet::Sunray(s) => {
81                if let Some(index) = state.cells_iter().position(|cell| !cell.is_charged()) {
82                    let cell = state.cell_mut(index);
83                    cell.charge(s);
84                    match state.build_rocket(index) {
85                        Ok(()) => println!("Rocket built successfully"),
86                        Err(e) => println!("Rocekt Failed to be built: {e}"),
87                    }
88                }
89                Some(SunrayAck {
90                    planet_id: state.id(),
91                })
92            }
93            OrchestratorToPlanet::InternalStateRequest => Some(InternalStateResponse {
94                planet_id: state.id(),
95                planet_state: state.to_dummy(),
96            }),
97            OrchestratorToPlanet::OutgoingExplorerRequest { .. }
98            | OrchestratorToPlanet::IncomingExplorerRequest { .. }
99            | OrchestratorToPlanet::Asteroid(_)
100            | OrchestratorToPlanet::StartPlanetAI
101            | OrchestratorToPlanet::StopPlanetAI => None,
102        }
103    }

105    /// Handles incoming messages from an `Explorer` agent and generates appropriate responses based on the planet's current state.
106    ///
107    /// This function processes various types of requests such as resource availability, combination support,
108    /// generation, and energy cell status. If the planet is stopped (shut down or inactive), no responses are sent.
109    ///
110    /// # Parameters
111    ///
112    /// * `self`: Mutable reference to the planet's controller or handler, which includes runtime state like `is_stopped`.
113    /// * `state`: Mutable reference to the current `PlanetState`, providing access to data like energy cells, resources, etc.
114    /// * `generator`: reference for `Generator`.
115    /// * `comb`: reference for `Combinator`.
116    /// * `msg`: The incoming message from the explorer, wrapped in the `ExplorerToPlanet` enum.
117    ///
118    /// # Returns
119    ///
120    /// Returns an `Option<PlanetToExplorer>`:
121    /// - `Some(response)` if a valid response can be generated.
122    /// - `None` if the planet is stopped or the message type is unsupported/not yet implemented.
123    ///
124    /// # Message Handling
125    ///
126    /// Currently supports:
127    /// - `AvailableEnergyCellRequest`: Responds with the count of charged energy cells.
128    /// - `SupportedCombinationRequest`: Respond with the list of available comination recipes so
129    ///   an empty hashset
130    /// - `CombineResourceRequest`: Responde with the complex rescourc this planet can generate so
131    ///   `None`
132    /// - `SupportedResourceRequest`: Responds with the basic resource type hashset containing the
133    ///   only supported resource `Oxygen`
134    /// - `GenerateResourceRequest`: Responds only to request for the `Oxygen` resource althought
135    ///   return `None`
136    ///
137    /// # Panics
138    ///
139    /// Panics if a non-implemented message variant is received.
140    fn handle_explorer_msg(
141        &mut self,
142        state: &mut PlanetState,
143        generator: &Generator,
144        comb: &Combinator,
145        msg: ExplorerToPlanet,
146    ) -> Option<PlanetToExplorer> {
147        if self.is_stopped {
148            return None;
149        }
150        match msg {
151            ExplorerToPlanet::SupportedResourceRequest { explorer_id: _ } => {
152                Some(PlanetToExplorer::SupportedResourceResponse {
153                    resource_list: generator.all_available_recipes(),
154                })
155            }
156            ExplorerToPlanet::GenerateResourceRequest {
157                explorer_id: _,
158                resource: BasicResourceType::Oxygen,
159            } => state
160                .cells_iter()
161                .position(EnergyCell::is_charged)
162                .and_then(|index| generator.make_oxygen(state.cell_mut(index)).ok())
163                .map(|r| {
164                    println!("Resource generated");
165                    PlanetToExplorer::GenerateResourceResponse {
166                        resource: Some(common_game::components::resource::BasicResource::Oxygen(r)),
167                    }
168                }),
169            ExplorerToPlanet::GenerateResourceRequest { .. } => None,
170            ExplorerToPlanet::SupportedCombinationRequest { .. } => {
171                Some(PlanetToExplorer::SupportedCombinationResponse {
172                    combination_list: comb.all_available_recipes(),
173                })
174            }
175            ExplorerToPlanet::CombineResourceRequest { .. } => {
176                /*Some(PlanetToExplorer::CombineResourceResponse {
177                    complex_response: None,
178                })*/
179                None
180            }
181            ExplorerToPlanet::AvailableEnergyCellRequest { .. } => {
182                let tmp = state.cells_iter().filter(|&cell| cell.is_charged()).count();
183                let count = tmp.try_into().unwrap_or_default();
184                Some(PlanetToExplorer::AvailableEnergyCellResponse {
185                    available_cells: count,
186                })
187            }
188        }
189    }

191    /// Handles an incoming asteroid event by launching an existing rocket or building a new one.
192    ///
193    /// # Behavior
194    ///
195    /// 1. **Launch**: If a rocket is already built (`state.has_rocket()`), it is launched immediately
196    ///    and returned.
197    /// 2. **Build & Launch**: If no rocket exists, the method searches for the first charged energy cell
198    ///    and attempts to build a rocket on it. If successful, the newly built rocket is launched and returned.
199    /// 3. **Failure**: Returns `None` if no rocket was available and construction failed or no charged cell existed.
200    ///
201    /// # Returns
202    ///
203    /// - `Some(Rocket)`: A rocket was successfully launched (either pre-existing or newly built).
204    /// - `None`: No rocket was launched (no rocket present and build failed or no charged cell).
205    ///
206    /// # Side Effects
207    ///
208    /// - Mutates `state`: may consume a rocket via `take_rocket()` and modify cells during construction.
209    /// - Prints log messages on build success or failure (consider using `log` crate instead of `println!`).
210    ///
211    fn handle_asteroid(
212        &mut self,
213        state: &mut PlanetState,
214        _: &Generator,
215        _: &Combinator,
216    ) -> Option<Rocket> {
217        if state.has_rocket() {
218            return state.take_rocket();
219        }
220        if let Some(index) = state.cells_iter().position(EnergyCell::is_charged) {
221            match state.build_rocket(index) {
222                Ok(()) => {
223                    println!("Rocket built successfully");
224                    return state.take_rocket();
225                }
226                Err(e) => println!("Rocekt Failed to be built: {e}"),
227            }
228        }
229        None
230    }
231}
